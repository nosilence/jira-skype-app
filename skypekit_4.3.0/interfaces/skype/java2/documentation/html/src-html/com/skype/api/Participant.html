794            protected void sidOnChangedProperty(final int propertyId, final int value, final String svalue) {
795                    final Property property = Property.get(propertyId);
796                    if (property == Property.P_UNKNOWN) return;
797                    final int idx = property.getIdx();
798                    if (idx != 0) {
799                            int bit  = 1<<((idx-1)%32);
800                            synchronized (this) {
801                                    mSidCached|=bit;
802                                    switch (propertyId) {
803                                    case 930:
804                                            if (value != 0)
805                                                    mConversation = (Conversation) mSidRoot.sidGetObject(property.getModuleId(), value);
806                                            else {
807                                                    mConversation = null;
808                                                    mSidCached &=~bit;
809                                            }
810                                            break;
811                                    case 931:
812                                            if (svalue != null) mIdentity = svalue;
813                                            else mSidCached &=~bit;
814                                            break;
815                                    case 932: mRank = Rank.get(value); break;
816                                    case 933: mRequestedRank = Rank.get(value); break;
817                                    case 934: mTextStatus = TextStatus.get(value); break;
818                                    case 935: mVoiceStatus = VoiceStatus.get(value); break;
819                                    case 936: mVideoStatus = VideoStatus.get(value); break;
820                                    case 943:
821                                            if (svalue != null) mLiveIdentity = svalue;
822                                            else mSidCached &=~bit;
823                                            break;
824                                    case 938:
825                                            if (svalue != null) mLivePriceForMe = svalue;
826                                            else mSidCached &=~bit;
827                                            break;
828                                    case 948:
829                                            if (svalue != null) mLiveFwdIdentities = svalue;
830                                            else mSidCached &=~bit;
831                                            break;
832                                    case 941: mSoundLevel = value; break;
833                                    case 942:
834                                            if (svalue != null) mDebugInfo = svalue;
835                                            else mSidCached &=~bit;
836                                            break;
837                                    case 947:
838                                            if (svalue != null) mLastVoiceError = svalue;
839                                            else mSidCached &=~bit;
840                                            break;
841                                    case 949:
842                                            if (svalue != null) mQualityProblems = svalue;
843                                            else mSidCached &=~bit;
844                                            break;
845                                    case 950: mLiveType = Skype.IdentityType.get(value); break;
846                                    case 951:
847                                            if (svalue != null) mLiveCountry = svalue;
848                                            else mSidCached &=~bit;
849                                            break;
850                                    case 952:
851                                            if (svalue != null) mTransferredBy = svalue;
852                                            else mSidCached &=~bit;
853                                            break;
854                                    case 953:
855                                            if (svalue != null) mTransferredTo = svalue;
856                                            else mSidCached &=~bit;
857                                            break;
858                                    case 954:
859                                            if (svalue != null) mAdder = svalue;
860                                            else mSidCached &=~bit;
861                                            break;
862                                    case 955: mLastLeavereason = Skype.LeaveReason.get(value); break;
863                                    default: mSidCached&=~bit; break;
864                                    }
865                            }
866                    }
867                    ParticipantListener listener = ((Skype) mSidRoot).getParticipantListener();
868                    if (listener != null)
869                            listener.onPropertyChange(this, property, value, svalue);
870            }

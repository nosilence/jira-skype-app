049            public void skipBytes(int numBytes) throws IOException {
050                    if (mApplicationDataToRead == null)
051                            throw new  IOException("Connection was closed");
052    
053                    while (numBytes > mAvailable) {
054                            numBytes  -= mAvailable;
055                            mApplicationDataToRead.clear();
056                            mAvailable = read(mApplicationDataToRead);
057                    }
058    
059                    mAvailable -= numBytes;
060                    if (mAvailable > 0) {
061                            mApplicationDataToRead.position(mApplicationDataToRead.position()+numBytes);
062                    } else {
063                            mApplicationDataToRead.clear();
064                    }
065            }

067            public void readBytes(byte[] dest) throws IOException {
068                    readBytes(dest, 0, dest.length);
069            }

071            private int read(ByteBuffer dest) throws IOException {
072                    // assert dest.capacity() == ApplicationBufferSize
073                    int produced = 0;
074                    int rd = 0;
075                    if (mPacketToRead.position() == 0) { // need to fill,  packet shall be clear
076                            rd = mInput.read(mPacketToRead.array(), 0, mPacketBufferSize);
077                            if (rd < 0) throw new IOException("Connection closed");
078                            mPacketToRead.limit(rd);
079                    } else {
080                            mPacketToRead.flip();
081                    }
082    //              boolean cont = true; 
083    //              while (cont) {
084                            SSLEngineResult result;
085                            try {
086                                    result = mEngine.unwrap(mPacketToRead, dest);
087                            } catch (java.lang.Exception e) {
088                                    throw new IOException("error when unwrapping " + e.getMessage());
089                            }               
090                            produced += result.bytesProduced();
091                            switch (result.getStatus()) {
092                            case OK:
093                                    if (mPacketToRead.remaining() == 0) {
094                                            mPacketToRead.clear();
095    //                                      cont = false;
096                                            break;
097                                    }
098                                    mPacketToRead.compact();
099    //                              mPacketToRead.flip();
100                                    break;
101                            case BUFFER_UNDERFLOW:
102    //                              cont = false; // better flushing what has been produced, to avoid overflow as we may already have decoded
103                                                  // quite some packets
104                                    int offset = mPacketToRead.remaining();
105                                    if (offset > 0) {
106                                            // there are some data left, prefill for the next read call as we must force input read
107                                            rd = mPacketBufferSize - offset;
108                                            mPacketToRead.compact();
109                                            rd = mInput.read(mPacketToRead.array(), offset, rd);
110                                            if (rd < 0) throw new IOException("Connection closed");
111                                            mPacketToRead.position(offset+rd);
112                                    } else {
113                                            mPacketToRead.clear();
114                                    }
115                                    break;
116                            default:
117                                    // assumption a PacketBufferSize can fill at most an ApplicationBufferSize, thus no overflow expected: => ApplicationBufferSize == 0 when reading...
118                                    throw new IOException("TLS "+result.getStatus() + " error while reading "+mPacketToRead+" dest:"+dest);
119                            }
120    //              }
121                    if (mInputTransportLog != null && produced > 0) {
122                            mInputTransportLog.write(dest.array(), 0, produced);
123                    }
124                    return produced;
125            }

127            public void readBytes(byte[] dest, int offset, int length) throws IOException {
128                    if (mApplicationDataToRead == null)
129                            throw new  IOException("Connection was closed");
130                    if (length > mAvailable) {
131    
132                            if (mAvailable > 0) {
133                                    mApplicationDataToRead.get(dest, 0, mAvailable);
134                                    length -= mAvailable;
135                                    offset  = mAvailable;
136                                    mApplicationDataToRead.clear();
137                            }
138    
139                            while (length >= mApplicationBufferSize) {
140                                    ByteBuffer dst = ByteBuffer.wrap(dest, offset, length);
141                                    int rd = read(dst);
142                                    length -= rd;
143                                    offset += rd;
144                            }
145    
146                            while (length > 0) {
147                                    int rd = read(mApplicationDataToRead);
148                                    mApplicationDataToRead.position(0);
149                                    mApplicationDataToRead.limit(rd);
150    
151                                    if (rd >= length) {
152                                            mAvailable = rd;
153                                            break;
154                                    }
155    
156                                    mApplicationDataToRead.get(dest, offset, rd);
157                                    mApplicationDataToRead.clear();
158                                    offset += rd;
159                                    length -= rd;
160                            }
161    
162                    }
163    
164                    mApplicationDataToRead.get(dest, offset, length);
165                    mAvailable -=  length;
166                    if (mAvailable == 0)
167                            mApplicationDataToRead.clear();
168    //              for (int i = 0, e = dest.length; i < e; i++)
169    //              System.err.println( "'" + dest[i] + "'" );
170    
171            }

173            public int readByte() throws IOException {
174                    if (mApplicationDataToRead == null)
175                            throw new  IOException("Connection was closed");
176                    while (mAvailable == 0) {
177                            mAvailable = read(mApplicationDataToRead);
178                            if (mAvailable > 0)
179                                    mApplicationDataToRead.flip();
180                    }
181                    byte b = mApplicationDataToRead.get();
182    //          System.err.println( "'" + b + "'" );
183                    mAvailable--;
184                    if (mAvailable == 0)
185                            mApplicationDataToRead.clear();
186                    return b;
187            }

189            public void writeBytes(byte [] src) throws IOException {
190                    if (mApplicationDataToSend == null) throw new IOException("Connection was closed");
191    
192                    int length = src.length;
193                    if (mOutputTransportLog != null && length > 0) {
194                            mOutputTransportLog.write(src, 0, length);
195                    }
196                    if ((length+mApplicationDataSize) > mApplicationBufferSize) { // use some gather call?
197                            if (mApplicationDataSize > 0) { 
198                                    write(mApplicationDataToSend);
199                                    mApplicationDataSize = 0;
200                            }
201                            ByteBuffer buf = ByteBuffer.wrap(src, 0, length);
202                            buf.position(buf.limit());
203                            write(buf);
204                            return;
205                    }
206                    mApplicationDataToSend.put(src, 0, length);
207                    mApplicationDataSize += length; // getPosition shall be the same...
208            }

210            public void writeBytesAndFlush(byte [] src) throws IOException {
211                    writeBytes(src);
212                    if (mApplicationDataSize > 0) {
213                            write(mApplicationDataToSend);
214                            mApplicationDataSize = 0;
215                    }
216            }

218            public void write(ByteBuffer src) throws IOException {
219                    src.flip();
220                    SSLEngineResult result;
221                    do {
222                            try {
223                                    result = mEngine.wrap(src, mPacketToSend);
224                                    if (result.getStatus() != SSLEngineResult.Status.OK) throw new IOException("error when wrapping");
225                            } catch (SSLException e) {
226                                    throw new IOException("error when wrapping " + e.getMessage());
227                            }
228                            mOutput.write(mPacketToSend.array(), 0, result.bytesProduced());
229                            mPacketToSend.clear();
230                    } while (src.hasRemaining());
231                    src.clear();
232            }

234            public void writeByte(int value) throws IOException {
235                    if (mApplicationDataToSend == null) throw new IOException("Connection was closed");
236                    if (mApplicationDataSize == mApplicationBufferSize) {
237                            write(mApplicationDataToSend);
238                            mApplicationDataSize = 0;
239                    }
240                    mApplicationDataToSend.put((byte) value);
241                    if (mOutputTransportLog != null) {
242                            mOutputTransportLog.write((byte) value);
243                    }
244                    mApplicationDataSize++; // getPosition shall be the same...
245            }

247            public void writeByteAndFlush(int value) throws IOException {
248                    if (mApplicationDataToSend == null) throw new IOException("Connection was closed");
249                    if (mApplicationDataSize == mApplicationBufferSize) {
250                            write(mApplicationDataToSend);
251                    }
252                    mApplicationDataToSend.put((byte) value);
253                    write(mApplicationDataToSend); // hopefully works with 1 byte... else one may fill with useless 'zzz'
254                    if (mOutputTransportLog != null) {
255                            mOutputTransportLog.write((byte) value);
256                    }
257                    mApplicationDataSize = 0;
258            }

274            public void init(ClientConfiguration cfg) throws IOException
275            {
276                    try {
277                            context = SSLContext.getInstance("TLS");
278                    } catch (NoSuchAlgorithmException e) {
279                            throw new IOException("TLSServerTransport:IOException:" + e.getMessage() );
280                    }
281    
282                    dummyTrustManager trustManager = new dummyTrustManager();
283                    dummyKeyManager   keyManager   = new dummyKeyManager(cfg.getX509Certificate(),cfg.getPrivateKey());
284    
285                    KeyManager[] km =   { keyManager   };
286                    TrustManager[] tm = { trustManager };
287    
288                    try {
289                            context.init(km, tm, null);
290                    } catch (KeyManagementException e) {
291                            throw new IOException("TLSServerTransport:KeyManagementException:" + e.getMessage() );
292                    }
293    
294                    SSLEngine engine = context.createSSLEngine();
295                    engine.setUseClientMode(false);
296                    engine.setNeedClientAuth(false);
297                    engine.setWantClientAuth(false);
298    
299                    mEngine                 = engine;
300                    SSLSession s            = engine.getSession();
301                    mApplicationBufferSize  = s.getApplicationBufferSize();
302                    mPacketBufferSize       = s.getPacketBufferSize();
303                    mPacketToSend           = ByteBuffer.allocate(mPacketBufferSize);
304                    mPacketToRead           = ByteBuffer.allocate(mPacketBufferSize);
305                    mApplicationDataToSend  = ByteBuffer.allocate(mApplicationBufferSize); // may rather use the configuration one
306                    mApplicationDataToRead  = ByteBuffer.allocate(mApplicationBufferSize);
307    
308                    try {
309                            @SuppressWarnings("rawtypes")
310                            Class c = Class.forName("javax.net.ssl.SSLEngine");
311                            for (Method m : c.getDeclaredMethods() ) {
312                                    if (m.getName() == "getSSLParameters") { //this is not supported on Android < 2.3
313                                            SSLParameters params = engine.getSSLParameters();
314                                            params.setCipherSuites(new String [] {
315                                                    "TLS_RSA_WITH_AES_128_CBC_SHA"
316                                            });
317                                            engine.setSSLParameters(params);
318                                            }
319                                    }
320                            }
321                    catch (ClassNotFoundException e) { //shouldnt ever happen
322                        throw new IOException("TLSServerTransport:init_failed:" + e.getMessage(), e);
323                    }
324    
325                    try {
326                            do_handshake();
327                    }
328                    catch (SSLException e) {
329                        throw new IOException("TLSServerTransport:handshake_failed:" + e.getMessage(), e);
330                    }
331                    catch (IOException e) {
332                        throw new IOException("TLSServerTransport:handshake_failed:" + e.getMessage(), e);
333                    }
334            }
335    
336            void do_handshake() throws SSLException, IOException
337            {
338    
339                    SSLEngine engine = mEngine;
340                    HandshakeStatus result =  HandshakeStatus.NEED_UNWRAP;
341                    while(true) {
342                        switch(result) {
343                        case FINISHED:
344                            return;
345                        case NEED_TASK:
346                            Runnable task;
347                            while((task=engine.getDelegatedTask()) != null) {
348                                task.run();
349                            }
350                            break;
351                        case NEED_UNWRAP: {
352                            mAvailable = read(mApplicationDataToRead); // shall be 0...
353                            break;
354                        }
355                        case NEED_WRAP: {
356                            write(mApplicationDataToSend); // shall be empty...
357                            break;
358                        }
359                        case NOT_HANDSHAKING:
360                            return;
361                        }
362                        result = engine.getHandshakeStatus();
363    
364            }
365            }

367            public void close() throws IOException {
368                    mPacketToSend           = null;
369                    mPacketToRead           = null;
370                    mApplicationDataToSend  = null; // may rather use the configuration one
371                    mApplicationDataToRead  = null;
372                    try {
373                            if (mInputTransportLog != null)
374                                    mInputTransportLog.close();
375                    } catch (IOException e) {
376                    }
377                    mInputTransportLog = null;
378                    try {
379                            if (mOutputTransportLog != null)
380                                    mOutputTransportLog.close();
381                    } catch (IOException e) {
382                    }
383                    mOutputTransportLog = null;
384                    mInput.close();
385                    mOutput.close();
386            }

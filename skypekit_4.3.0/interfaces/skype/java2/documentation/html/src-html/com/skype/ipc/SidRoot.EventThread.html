042        protected void sidPollEvent() {
043            try {
044                while (!mSidStopped) {
045                    Decoding decoder = mSidDecoder.decodeEvent();
046                    switch (decoder.getCommand()) {
047                    case 'C': 
048    //System.out.println("### sidPollEvent(): sidOnChangedProperty "+(decoder==mSidDecoder));
049                        sidOnChangedProperty(decoder);
050                        break;
051                    case 'E':
052    //System.out.println("### sidPollEvent(): sidOnChangedProperty "+(decoder==mSidDecoder));
053                        int mid  = decoder.decodeUint();
054                        int evid = decoder.decodeUint();
055                        sidDispatchEvent(mid, evid, decoder);
056                        break;
057                    case 'g':
058                        sidAddPendingGetResponse();
059                        break;
060                    case 'r':
061                        sidAddPendingResponse(decoder.decodeUint());
062                        break;
063                    }
064                }
065            } catch (IOException e) {
066                sidOnFatalError(e);
067            }
068        }

072        /***
073         * initialize the connection to the runtime, and the interface cache accordingly to the configuration
074         * @param configuration defining the connection parameters to the runtime
075         * @param listener callbacks to check the status of the connection
076         * return success or failure to connect
077         */
078        public boolean init(ClientConfiguration configuration, ConnectionListener listener) {
079            registerConnectionListener(listener);
080    
081            TransportFactory factory = configuration.getTransportFactory();
082            TransportFactory.Result transport = factory.init(configuration, listener);
083            if (transport != null) {
084                mSidInput           = transport.in;
085                mSidOutput          = transport.out;
086                mSidDispatchAll     = configuration.isDispatchAll();
087                mSidDecoder         = new BinProtocolClientDecoder(transport.in, this, this);
088                mSidEncoder         = new BinProtocolClientEncoder(transport.out, this);
089                flushCache(null);
090                mSidEncoderLock     = new ReentrantLock();
091                mSidEncoding        = false;
092                mSidPendingRequests = new HashMap<Integer, Thread>(2);
093                mSidPendingOneWayRequests = new HashSet<Integer>(2);
094                mSidPendingGets     = new ArrayDeque<Thread>(2);
095                mSidEventThread     = new EventThread(this);
096                mSidReader          = mSidEventThread;
097                mSidPrevReader      = null;
098                mSidEventThread.start();
099            }
100            return mSidObjects != null;
101        }

103        protected SidObject sidDecodeEventTarget(int mid, Decoding decoder)
104        {
105            SidObject target = null;
106            try {
107                int b = decoder.decodeTag(); // shall be 'O'
108                int zero = decoder.decodeUint();
109                int oid = decoder.decodeUint();
110                target = mSidDispatchAll ? sidGetObject(mid, oid) : sidGetObjectIfPresent(mid, oid);
111                if (target == null) {
112                    decoder.skipMessage();
113                }
114            }
115            catch (IOException e) {
116                sidOnFatalError(e);
117            }
118            return target;
119        }

129        public synchronized boolean stop() {
130            if (mSidEncoding) {
131                try {
132                    mSidEncoderLock.unlock();
133                    mSidEncoding = false;
134                } catch (java.lang.IllegalMonitorStateException e) {
135                }
136            }
137            if (mSidStopped) return true;
138            mSidStopped = true;
139            try {
140                if (mSidInput != null) mSidInput.close();
141            } catch (IOException e) {
142            }
143            try {
144                if (mSidOutput != null) mSidOutput.close();
145            } catch (IOException e) {
146            }
147            if (mSidPrevReader != null)
148                mSidPrevReader.interrupt(); 
149            if (mSidReader != null && mSidReader != mSidPrevReader)
150                mSidReader.interrupt(); 
151            if (mSidPendingRequests != null)
152                for (Thread thread : mSidPendingRequests.values()) {
153                    thread.interrupt();
154                }
155            if (mSidPendingGets != null)
156                for (Thread thread : mSidPendingGets) {
157                    thread.interrupt();
158                }
159            return false;
160        }

162        public void sidOnFatalError(IOException e) {
163            if (!stop()) {
164                if (mSidConnectionListener != null) {
165                    StringWriter sw = new StringWriter();
166                    e.printStackTrace(new PrintWriter(sw));
167                    mSidConnectionListener.sidOnDisconnected(sw.toString());
168                } else {
169                    e.printStackTrace();
170                    System.exit(-1);
171                }
172            }// else
173    //                e.printStackTrace();
174        }

180        public SidObject sidGetObjectIfPresent(int mid, int oid) {
181            Integer key = oid; //new Integer(oid);
182            SidObject o = null;
183            synchronized (mSidObjects) {
184                SoftReference<SidObject> r = mSidObjects.get(key); 
185                if (r != null) {
186                    o = r.get();
187                    if (o == null) {
188                        o = sidCreateObject(mid, oid);
189                        mSidObjects.put(key, new SoftReference<SidObject>(o));
190                    }
191                }
192            }
193            return o;
194        }

243        public Decoding sidOnGetRequestEncoded() throws IOException {
244            // current thread has posted a get request and no other can post one for now
245            Thread current   = Thread.currentThread();
246            boolean iamreader = false;
247            boolean responded = false;
248            synchronized (this) {
249                responded = mSidPendingGetResponse;
250                if (responded) { 
251                    // assert(mSidPendingGets.size() == 0 && mSidPrevReader == mSidPendingGetResponse);
252                    // my response is already here... and was catched by mSidPrevReader
253                    // we will read it, and resume mSidReader once we have read it
254                    mSidPendingGetResponse = false;
255                    mSidReader = current;
256    //System.out.println("### sidOnGetRequestEncoded, reader already got my answer "+current+ " Prev:"+mSidPrevReader);
257                } else {
258                    // add us to the get request pending queue 
259                    mSidPendingGets.add(current); 
260                    // if no one is reading, current is promoted as reader
261                    if (mSidReader == null) {
262                        mSidReader = current;
263                        iamreader  = true;
264    //System.out.println("### sidOnGetRequestEncoded, no reader electing myself "+current);
265                    }
266                }
267            }
268            mSidEncoding = false;
269            mSidEncoderLock.unlock();
270            // we released the encoding lock...  there can be some race condition here,
271            // but if the response is here, or the current thread is promoted reader
272            if (!responded) {
273                // the response wasn't here, we have to wait for it
274                if (!iamreader) {
275                    // someone else is reading, we have to wait to be waken up,
276                    // either because no one reads anymore, or because mSidReader
277                    // catched our response
278                    synchronized (current) {
279                        try {
280                            while (mSidReader != current)
281                                current.wait();
282                            iamreader = mSidPrevReader == null;
283                        } catch (InterruptedException ie) {
284                            mSidReader = null;
285                        }
286                    }
287                }
288                if (iamreader) {
289                    // current thread became the new reader and shall wait for a response
290                    sidWaitResponse();
291                }
292            }
293            // response is here, mine and needs being decoded (and once decoded, we must resume mSidReader)
294            return mSidDecoder;
295        }

297        public Decoding sidOnRequestEncoded(int requestId) throws IOException {
298            mSidEncoding = false;
299            mSidEncoderLock.unlock();
300            Thread current = Thread.currentThread();
301            boolean iamreader = false;
302            Integer rid    = requestId;
303            synchronized (this) {
304                Thread reader = mSidPendingRequests.get(rid);
305                if (reader != null) {
306                    // response already here and was catched by reader
307                    // assert(reader == mSidReader == mSidPrevReader);
308                    mSidPendingRequests.remove(rid);
309                    mSidReader = current; 
310    //System.out.println("### sidOnRequestEncoded "+requestId+", reader already got my answer "+current+ " Prev:"+mSidPrevReader);
311                    return mSidDecoder;
312                } else
313    //System.out.println("### sidOnRequestEncoded"+requestId+", by "+current);
314                mSidPendingRequests.put(rid, current);
315                if (mSidReader == null) {
316    //System.out.println("### sidOnRequestEncoded"+requestId+", no reader electing myself "+current);
317                    mSidReader = current;
318                    iamreader = true;
319                }
320            }
321            if (!iamreader) {
322                synchronized (current) {
323                    try {
324                        while (mSidReader != current)
325                            current.wait();
326                        // 2 cases => either there was no more reader and we were promoted reader
327                        //         => or the reader got our response, we must decode it
328                        iamreader = mSidPrevReader == null;
329                    } catch (InterruptedException ie) {
330                        mSidReader = null;
331                    }
332                }
333            }
334            if (iamreader) { // we were wakenup to read or there was no reader
335                sidWaitResponse();
336            }
337            return mSidDecoder;
338        }

340        public Decoding sidOnOneWayRequestEncoded(int requestId) {
341            mSidEncoding = false;
342            mSidEncoderLock.unlock();
343    
344            Thread responder = null;
345            Thread current   = Thread.currentThread();
346            Integer rid      = requestId;
347    
348            synchronized (this) {
349                responder = mSidPendingRequests.get(rid);
350                if (responder != null) { 
351                    // the response is already here and was read by the thread "mSidPrevReader"
352                    // the responder is waiting that we are parsing our response to resume
353                    // reading
354                    // assert(mSidPrevReader != null && mSidPrevReader == mSidReader)
355                    mSidPendingRequests.remove(rid);
356                    mSidReader = current; 
357    //System.out.println("### sidOnOneWayRequestEncoded"+requestId+", read signaled response, taking reading "+current+" prev:"+mSidPrevReader);
358                } else { 
359                    // the response is not here, mark it as one way so that the reader thread
360                    // can skip it
361                    mSidPendingRequests.put(rid, current); 
362                    mSidPendingOneWayRequests.add(rid);
363    //System.out.println("### sidOnOneWayRequestEncoded"+requestId+", by "+current);
364                }
365            }
366    
367            if (responder != null) {
368                // so, another thread catched our response and didn know it was one way,
369                // thus it notified us to read it
370                // we must ignore the rest of the response and tell the other thread
371                // it can resume reading
372                try {
373    //System.out.println("### sidOnOneWayRequestEncoded"+requestId+", skipping "+current);
374                    mSidDecoder.skipMessage();
375                } catch (IOException e) {
376                    sidOnFatalError(e);
377                }
378            }
379    
380            return null;
381        }

383        void sidOnChangedProperty(Decoding decoder) {
384            // currently only 1 notified at once and only int-alike properties are valued, else design shall be changed to ensure that all properties are decoded
385            // before calling the callbacks
386            int expected = 1;
387            int kind;
388            int oid       = 0;
389            int moduleId  = 0;
390            int value     = 0;
391            String svalue = null;
392            int propertyId= -1;
393            try {
394                while (decoder.hasNextProperty()) {
395                    assert(expected-- > 0);
396                    PropertyInfo info = decoder.getNextProperty();
397                    kind       = info.kind; 
398                    moduleId   = info.moduleId;
399                    oid        = info.objectId;
400                    propertyId = info.propertyId;
401                    switch (kind) {
402                    case 'F': case 'N': break;
403                    case 'T': value = 1; break;
404                    case 'O': case 'e': case 'u': value = decoder.decodeUint(); break;
405                    case 'i': value = decoder.decodeInt(); break;
406                    case 'S': case 'f': case 'X': svalue =  decoder.decodeString(); break;
407                    default:
408                        decoder.skipValue(kind);
409                        break;
410                    }
411                }
412            } catch (IOException e) {
413                sidOnFatalError(e);
414            }
415            if (oid>0) {
416                SidObject o = mSidDispatchAll ? sidGetObject(moduleId, oid) : sidGetObjectIfPresent(moduleId, oid);
417                if (o != null) {
418                    o.sidOnChangedProperty(propertyId, value, svalue);
419                }
420            }
421        }

423        public boolean sidWantRead() {
424            synchronized (this) {
425               if ((mSidReader == null || mSidReader == mSidEventThread) && mSidPrevReader == null) {
426    //System.out.println("### sidWantRead() elected because "+mSidReader);
427                     mSidReader = mSidEventThread;
428                     return true;
429                } else {
430    //System.out.println("### sidWantRead() event thread ready ");
431                    mSidReadyEventThread = true;
432                }
433            }
434            synchronized (mSidEventThread) { // we may have been elected reader while locking on this
435                try {
436                    while (mSidReader != mSidEventThread && mSidReadyEventThread)
437                        mSidEventThread.wait();
438    //System.out.println("### sidWantRead() running "+mSidReader);
439                } catch (InterruptedException ie) {
440                    return false;
441                }
442            }
443            return true;
444        }

461        public void sidOnMessageDecoded() {
462            Thread newReader = null;
463            synchronized (this) {
464                newReader      = mSidPrevReader;
465                mSidPrevReader = null;
466                if (newReader == null) {
467                    // elect a requester if any, provided that it waits for a response
468                    if (!mSidPendingGets.isEmpty()) {
469                        newReader = mSidPendingGets.peek(); 
470                    } else if (!mSidPendingRequests.isEmpty()) {
471                        // can be oneway... then better ignoring
472                        for (Entry<Integer, Thread> entry : mSidPendingRequests.entrySet()) {
473                            Integer rid = entry.getKey();
474                            if (!mSidPendingOneWayRequests.contains(rid)) {
475                                newReader = entry.getValue();
476                                break;
477                            }
478                        }
479                    }
480                    // else take the event listener if it is ready
481                    if (newReader == null && mSidReadyEventThread) {
482                        newReader = mSidEventThread; 
483                        mSidReadyEventThread = false;
484                    }
485                } else if (mSidReader == newReader) {
486    //System.out.println("### sidOnMessageDecoded, mSidReader == newReader ");
487                    return;
488                }
489    //System.out.println("### sidOnMessageDecoded, elected "+newReader);
490                mSidReader = newReader;
491            }
492            if (newReader != null) {
493                synchronized (newReader) {
494                    newReader.notify();
495                }
496            }
497        }

499        boolean sidAddPendingGetResponse()  throws IOException {
500            Thread current   = mSidReader;
501            Thread requester = null;
502            boolean got_my_response = false;
503            synchronized (this) {
504                if (mSidPendingGets.isEmpty()) {
505                    mSidPendingGetResponse = true;
506                } else {
507                    requester = mSidPendingGets.pop();
508                    got_my_response = requester == current;
509                }
510    //System.out.println("### sidAddPendingGetResponse(): curr"+current+" known poster, "+requester);
511                if (!got_my_response) {
512                    mSidPrevReader = current; // tell the requester to wake us up after
513                    mSidReader = requester;
514                }
515            }
516            if (!got_my_response) {
517                if (requester != null) {
518                    // wake up the requester
519                    synchronized (requester) {
520                        requester.notify();
521                    }
522                }
523                // wait that the requester wakes us up
524                synchronized (current) {
525                    try {
526                        while (mSidReader != current)
527                            current.wait();
528                    } catch (InterruptedException ie) {
529                        throw new IOException("Connection was closed");
530                    }
531                }
532            }
533            return got_my_response;
534        }

536        boolean sidAddPendingResponse(int requestId)  throws IOException {
537            Thread current   = mSidReader;
538            Thread requester = null;
539            Integer rid      = requestId;
540            boolean oneway   = false;
541            boolean got_my_response = false;
542            synchronized (this) {
543                requester = mSidPendingRequests.get(rid);
544                if (requester == null) {
545                    // the requester hasn't yet waited for the response
546                    mSidPendingRequests.put(rid, current);
547                } else {
548                    mSidPendingRequests.remove(rid);
549                    oneway = mSidPendingOneWayRequests.remove(rid);
550                }
551    //System.out.println("### sidAddPendingResponse() "+requestId+": curr"+current+" known poster, "+requester+ " oneway="+oneway);
552                got_my_response = requester == current;
553                if (!got_my_response || oneway) {
554                    mSidPrevReader = current; // tell the requester to wake us up after
555                    if (!oneway)
556                        mSidReader = requester;
557                }
558            }
559            if (oneway) {
560                // we have found that the requester wasn't waiting for the response, skip it ourselves
561                try {
562                    mSidDecoder.skipMessage(); // will call OnMessageDecoded, but we want going on
563                } catch (IOException e) {
564                    sidOnFatalError(e);
565                }
566                return false;
567            }
568            if (!got_my_response) {
569                // tell the requester to read the response if we know who it is
570                if (requester != null) {
571                    synchronized (requester) {
572                        requester.notify();
573                    }
574                } else {
575                } 
576                // wait that the requester resumes us
577                synchronized (current) {
578                    try {
579                        while (mSidReader != current)
580                            current.wait();
581                    } catch (InterruptedException ie) {
582                        throw new IOException("Connection was closed");
583                    }
584                }
585            }
586            return got_my_response;
587        }

589        void sidWaitResponse() throws IOException {
590            Decoding decoder = mSidDecoder;
591            while (true) {
592                int r;
593                try {
594                    r = decoder.decodeResponse();
595                } catch (IOException e) {
596                     return;
597                }
598                switch (r) {
599                case 'r':
600                    int rid;
601                    try {
602                        rid = decoder.decodeUint();
603                    } catch (IOException e) {
604                        return;
605                    }
606                    if (sidAddPendingResponse(rid)) return;
607                    break;
608                case 'g':
609                    if (sidAddPendingGetResponse()) return;
610                    break;
611                }
612            }
613        }

625        public SidObject flushCache(SidObject keep) {
626            mSidTimestamp++;
627            mSidObjects = new HashMap<Integer, SoftReference<SidObject>>(512);
628            if (keep != null) {
629                mSidObjects.put(keep.getOid(), new SoftReference<SidObject>(keep));
630                keep.mSidTimestamp = mSidTimestamp;
631            }
632            return keep;
633        }
